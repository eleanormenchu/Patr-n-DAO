    public Connection EstablerConexion(){

            conectar = DriverManager.getConnection(cadena,usuario,contraseña);

        }
        
        catch (Exception e){
            System.out.println("Erro al conectar con la base de datos: " + e.toString());
        }
        
        return conectar;
        
    }
